## 三、核心数据源详细分析

### 3.1 nba_api (主推荐)

**简介**：nba_api是目前最推荐的NBA数据获取工具，它是Python开源库，直接封装了NBA官方stats.nba.com的API接口。

**官网/仓库**：
- GitHub: https://github.com/swar/nba_api
- 文档: https://nba-apidocumentation.knowledgeowl.com/

**核心优势**：
1. **数据权威**：直接对接NBA.com官方数据
2. **完全免费**：无需API Key
3. **功能全面**：100+个API端点
4. **易于使用**：Pythonic设计，返回Pandas DataFrame
5. **历史完整**：支持1947年至今所有赛季数据

**主要端点**：

```python
# 安装
pip install nba_api

# 获取球员职业生涯统计
from nba_api.stats.endpoints import playercareerstats
career = playercareerstats.PlayerCareerStats(player_id='2544')  # LeBron James
career.season_totals_regular_season.get_data_frame()

# 获取球员比赛日志
from nba_api.stats.endpoints import playergamelog
gamelog = playergamelog.PlayerGameLog(player_id='2544', season='2022-23')
gamelog.get_data_frames()[0]

# 获取比赛Box Score
from nba_api.stats.endpoints import boxscoretraditionalv2
box = boxscoretraditionalv2.BoxScoreTraditionalV2(game_id='0022400001')
box.get_data_frames()

# 获取今日比分
from nba_api.live.nba.endpoints import scoreboard
games = scoreboard.ScoreBoard()
games.get_dict()

# 获取球队信息
from nba_api.stats.static import teams, players
all_teams = teams.get_teams()  # 所有球队
lebron = players.find_players_by_full_name('LeBron James')  # 搜索球员
```

**适用场景**：球队近期状态、命中率、球员表现、比赛结果等特征工程所需数据。

### 5.2 历史比赛数据获取脚本

```python
"""
NBA历史比赛数据获取脚本
数据范围：2015-2026赛季
"""
from nba_api.stats.endpoints import leaguegamelog, teamgamelog
from nba_api.stats.static import teams
import pandas as pd
import time
import os

def get_season_games(season='2023-24', season_type='Regular Season'):
    """获取指定赛季所有比赛"""
    gamelog = leaguegamelog.LeagueGameLog(
        season=season,
        season_type_all=season_type
    )
    return gamelog.get_data_frames()[0]

def get_team_game_logs(team_id, season='2023-24'):
    """获取球队比赛日志（用于计算近期状态）"""
    gamelog = teamgamelog.TeamGameLog(
        team_id=team_id,
        season=season
    )
    return gamelog.get_data_frames()[0]

def collect_historical_data(start_year=2015, end_year=2026):
    """收集历史数据"""
    all_games = []
    
    for year in range(start_year, end_year + 1):
        season = f"{year}-{str(year+1)[-2:]}"
        print(f"正在获取 {season} 赛季数据...")
        
        try:
            # 获取常规赛数据
            regular_games = get_season_games(season, 'Regular Season')
            all_games.append(regular_games)
            
            # 获取季后赛数据
            playoff_games = get_season_games(season, 'Playoffs')
            all_games.append(playoff_games)
            
            # 避免请求过快
            time.sleep(1)
            
        except Exception as e:
            print(f"获取 {season} 数据失败: {e}")
            continue
    
    # 合并所有数据
    if all_games:
        df = pd.concat(all_games, ignore_index=True)
        return df
    return pd.DataFrame()

# 执行数据收集
if __name__ == "__main__":
    print("开始收集NBA历史比赛数据...")
    
    # 收集2015-2026赛季数据
    historical_data = collect_historical_data(2015, 2026)
    
    # 保存数据
    output_path = "nba_games_2015_2026.csv"
    historical_data.to_csv(output_path, index=False)
    print(f"数据已保存至 {output_path}")
    print(f"共获取 {len(historical_data)} 条比赛记录")
```

### 5.3 特征工程数据提取脚本

```python
"""
NBA比赛预测特征提取
"""
from nba_api.stats.endpoints import (
    teamgamelog, 
    playerestimatedmetrics,
    teamperformance
)
import pandas as pd
import numpy as np

def calculate_team_recent_form(team_id, n_games=10):
    """计算球队近期状态（最近N场表现）"""
    gamelog = teamgamelog.TeamGameLog(team_id=team_id)
    df = gamelog.get_data_frames()[0]
    
    # 转换WL为数值
    df['WIN'] = (df['WL'] == 'W').astype(int)
    
    # 计算近期战绩
    recent = df.head(n_games)
    return {
        'recent_wins': recent['WIN'].sum(),
        'recent_losses': len(recent) - recent['WIN'].sum(),
        'recent_win_rate': recent['WIN'].mean(),
        'recent_avg_pts': recent['PTS'].astype(float).mean(),
        'recent_avg_pts_against': recent['PTS_PA'].astype(float).mean()
    }

def get_home_away_stats(team_id):
    """获取主客场数据"""
    # 需要按主客场筛选数据
    gamelog = teamgamelog.TeamGameLog(team_id=team_id)
    df = gamelog.get_data_frames()[0]
    
    # 主场数据 (@ = 客场, vs = 主场)
    home_games = df[df['MATCHUP'].str.contains('vs.')]
    away_games = df[df['MATCHUP'].str.contains('@')]
    
    return {
        'home_wins': (home_games['WL'] == 'W').sum(),
        'home_losses': (home_games['WL'] == 'L').sum(),
        'away_wins': (away_games['WL'] == 'W').sum(),
        'away_losses': (away_games['WL'] == 'L').sum()
    }

def extract_game_features(home_team_id, away_team_id, game_date):
    """提取比赛特征"""
    # 球队近期状态
    home_form = calculate_team_recent_form(home_team_id)
    away_form = calculate_team_recent_form(away_team_id)
    
    # 主客场数据
    home_stats = get_home_away_stats(home_team_id)
    away_stats = get_home_away_stats(away_team_id)
    
    # 构建特征向量
    features = {
        # 基础统计
        'home_recent_win_rate': home_form['recent_win_rate'],
        'away_recent_win_rate': away_form['recent_win_rate'],
        'home_avg_pts': home_form['recent_avg_pts'],
        'away_avg_pts': away_form['recent_avg_pts'],
        
        # 主客场
        'home_team_home_win_rate': (
            home_stats['home_wins'] / 
            (home_stats['home_wins'] + home_stats['home_losses'] + 1e-6)
        ),
        'away_team_away_win_rate': (
            away_stats['away_wins'] / 
            (away_stats['away_wins'] + away_stats['away_losses'] + 1e-6)
        ),
    }
    
    return features

# 使用示例
if __name__ == "__main__":
    # 勇士 vs 湖人 示例
    features = extract_game_features(
        home_team_id=1610612744,  # Warriors
        away_team_id=1610612747,  # Lakers
        game_date='2024-03-15'
    )
    print("比赛特征:", features)
```

## 六、数据质量保障建议

### 6.1 速率控制
```python
import time
import random

def safe_request(func, max_retries=3, base_delay=2):
    """安全的请求包装器"""
    for attempt in range(max_retries):
        try:
            result = func()
            time.sleep(random.uniform(1, 3))  # 随机延迟1-3秒
            return result
        except Exception as e:
            if attempt < max_retries - 1:
                time.sleep(base_delay * (attempt + 1))
            else:
                raise e
```
